            let remaining = currentTargetUnixForHeader - now;
            if (remaining <= 0) {
                headerTimerEl.textContent = "ENDED!";
                if (headerIntervalId) {
                    clearInterval(headerIntervalId);
                }
                return;
            }
            const hours = String(Math.floor(remaining / 3600)).padStart(2, "0");

            headerTimerEl.textContent = `${hours}:${minutes}:${seconds}`;

        const nowForInitialCheckHeader = Math.floor(Date.now() / 1000);
        if (currentTargetUnixForHeader > nowForInitialCheckHeader) {
            updateHeaderTimer();
            headerIntervalId = setInterval(updateHeaderTimer, 1000);
        } else {
            headerTimerEl.textContent = "ENDED!";
        }
    } else {
        if(headerTimerEl) headerTimerEl.textContent = "Error";
    }

      const visibleGroups = Array.from(groups).filter(g => window.getComputedStyle(g).display !== 'none');
      if (visibleGroups.length >= 3) { 
        if(visibleGroups[0]) visibleGroups[0].setAttribute('data-label', 'Hours');
        if(visibleGroups[1]) visibleGroups[1].setAttribute('data-label', 'Minutes');
        if(visibleGroups[2]) visibleGroups[2].setAttribute('data-label', 'Seconds');
      }
    };
    
    const removeDividerDots = () => { 
      if (!mainFlipdownContainer) return;
      const dividers = document.querySelectorAll('#flipdown .rotor-group-divider');
      dividers.forEach(div => {
        div.textContent = '';
        div.style.display = 'none';
      });
    };

    setTimeout(() => {
      setLabels();
      removeDividerDots(); 
    }, 250); 
  }

  // PART 4: OTHER SCRIPTS (Card Scroller, View More Button)
  const cardsContainer = document.getElementById('cardsContainer');
  if (cardsContainer) {
    const scrollLeftBtn = document.getElementById('scrollLeft');
    const scrollRightBtn = document.getElementById('scrollRight');
    const firstCard = cardsContainer.querySelector('.card');
    const cardStyle = firstCard ? window.getComputedStyle(firstCard) : null;
    const cardMarginRight = cardStyle ? parseInt(cardStyle.marginRight) || 0 : 0; 
    const cardGap = 20; 
    const cardWidth = firstCard ? firstCard.offsetWidth + cardGap : 280; 
    const scrollAmount = cardWidth * 2;

    function updateArrows() {
      if (!cardsContainer || !scrollLeftBtn || !scrollRightBtn) return;
      scrollLeftBtn.style.display = cardsContainer.scrollLeft <= 10 ? 'none' : 'flex';
      scrollRightBtn.style.display =
        cardsContainer.scrollLeft + cardsContainer.clientWidth >= cardsContainer.scrollWidth - 10 ?
        'none' : 'flex';
    }

    if (scrollLeftBtn && scrollRightBtn) {
      scrollLeftBtn.addEventListener('click', () => {
        cardsContainer.scrollBy({ left: -scrollAmount, behavior: 'smooth' });
      });
      scrollRightBtn.addEventListener('click', () => {
        cardsContainer.scrollBy({ left: scrollAmount, behavior: 'smooth' });
      });
    }
    cardsContainer.addEventListener('scroll', updateArrows);
    window.addEventListener('resize', updateArrows);
    updateArrows();
  }

  const viewMoreButton = document.querySelector('.view-more-button-nembus');
  if (viewMoreButton) {
    viewMoreButton.addEventListener('click', function(e) {
      e.preventDefault();
      this.blur();
    });
  }
});
